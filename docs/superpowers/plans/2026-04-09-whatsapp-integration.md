# WhatsApp Integration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add WhatsApp as a platform integration to PingCRM, syncing personal chat history and contact profile info via a Node.js sidecar running whatsapp-web.js.

**Architecture:** A Node.js Express service (whatsapp-sidecar) wraps whatsapp-web.js and exposes REST + webhook APIs. The Python backend communicates with it over HTTP — initiating sessions, receiving real-time message events, and triggering backfills. Frontend shows a QR code flow for authentication.

**Tech Stack:** whatsapp-web.js (Node.js), Express, httpx (Python async HTTP), FastAPI, Celery, React/Next.js, qrcode.react

```python
"""WhatsApp integration helpers — contact matching and interaction upsert."""
import re
import uuid
from datetime import datetime

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.models.contact import Contact
from app.models.interaction import Interaction


def normalize_phone(phone: str) -> str:
    """Normalize a phone number to E.164 format.

    Strips formatting characters, removes @c.us suffix (WhatsApp JID),
    and ensures a leading '+'.
    """
    # Remove @c.us or @s.whatsapp.net suffix
    phone = re.sub(r"@[\w.]+$", "", phone)
    # Strip everything except digits and leading +
    digits = re.sub(r"[^\d]", "", phone)
    return f"+{digits}"


async def find_contact_by_whatsapp_phone(
    phone: str, user_id: uuid.UUID, db: AsyncSession
) -> Contact | None:
    """Find a contact by their whatsapp_phone field."""
    result = await db.execute(
        select(Contact)
        .where(Contact.user_id == user_id, Contact.whatsapp_phone == phone)
        .limit(1)
    )
    return result.scalar_one_or_none()


async def find_contact_by_phone_list(
    phone: str, user_id: uuid.UUID, db: AsyncSession
) -> Contact | None:
    """Find a contact whose phones array contains the given number."""
    result = await db.execute(
        select(Contact)
        .where(Contact.user_id == user_id, Contact.phones.contains([phone]))
        .limit(1)
    )
    return result.scalar_one_or_none()


async def resolve_contact(
    phone: str, user_id: uuid.UUID, db: AsyncSession, *, name: str | None = None
) -> tuple[Contact, bool]:
    """Find or create a contact for the given WhatsApp phone number.

    Lookup order:
    1. Match by whatsapp_phone
    2. Match by phones array
    3. Create new contact

    Returns (contact, is_new).
    """
    normalized = normalize_phone(phone)

    contact = await find_contact_by_whatsapp_phone(normalized, user_id, db)
    if contact:
        return contact, False

    contact = await find_contact_by_phone_list(normalized, user_id, db)
    if contact:
        # Link existing contact to WhatsApp
        contact.whatsapp_phone = normalized
        if name and not contact.full_name:
            contact.full_name = name
        return contact, False

    # Create new contact
    contact = Contact(
        id=uuid.uuid4(),
        user_id=user_id,
        full_name=name,
        whatsapp_phone=normalized,
        phones=[normalized],
        source="whatsapp",
    )
    db.add(contact)
    await db.flush()
    return contact, True


async def upsert_whatsapp_interaction(
    *,
    contact: Contact,
    user_id: uuid.UUID,
    message_id: str,
    direction: str,
    content_preview: str | None,
    occurred_at: datetime,
    db: AsyncSession,
) -> tuple[Interaction, bool]:
    """Create a WhatsApp interaction if it doesn't already exist.

    Returns (interaction, is_new).
    """
    result = await db.execute(
        select(Interaction).where(
            Interaction.raw_reference_id == message_id,
            Interaction.contact_id == contact.id,
            Interaction.user_id == user_id,
        ).limit(1)
    )
    existing = result.scalar_one_or_none()
    if existing:
        return existing, False

    interaction = Interaction(
        id=uuid.uuid4(),
        contact_id=contact.id,
        user_id=user_id,
        platform="whatsapp",
        direction=direction,
        content_preview=content_preview[:500] if content_preview else None,
        raw_reference_id=message_id,
        occurred_at=occurred_at,
    )
    db.add(interaction)
    return interaction, True
```

"use strict";

const crypto = require("node:crypto");

/**
 * Compute HMAC-SHA256 hex digest of a JSON string.
 */
function signPayload(body, secret) {
  return crypto.createHmac("sha256", secret).update(body).digest("hex");
}

/**
 * Create a webhook sender that POSTs signed events to the backend.
 *
 * @param {string} url - Backend webhook endpoint URL
 * @param {string} secret - HMAC signing secret
 * @returns {function} async send(event, data) function
 */
function createWebhookSender(url, secret) {
  async function send(event, data) {
    const payload = JSON.stringify({ event, data, timestamp: Date.now() });
    const signature = signPayload(payload, secret);

    try {
      const response = await fetch(url, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "X-Webhook-Signature": signature,
        },
        body: payload,
      });

      if (!response.ok) {
        console.error(
          JSON.stringify({
            level: "error",
            msg: "webhook_delivery_failed",
            event,
            status: response.status,
          })
        );
      }
    } catch (err) {
      console.error(
        JSON.stringify({
          level: "error",
          msg: "webhook_delivery_error",
          event,
          error: err.message,
        })
      );
    }
  }

  return send;
}

module.exports = { signPayload, createWebhookSender };
```

"use strict";

const { Client, LocalAuth } = require("whatsapp-web.js");
const config = require("./config");

/**
 * Manages WhatsApp client instances — one per userId.
 */
class SessionManager {
  constructor(webhookSend) {
    /** @type {Map<string, { client: Client, status: string, qr: string|null }>} */
    this.sessions = new Map();
    this.webhookSend = webhookSend;
  }

  /**
   * Start a new WhatsApp session for a user.
   * Returns QR code data string for scanning.
   */
  async startSession(userId) {
    if (this.sessions.size >= config.maxSessions) {
      throw new Error("Maximum session limit reached");
    }

    // Destroy existing session if any
    if (this.sessions.has(userId)) {
      await this.destroySession(userId);
    }

    const client = new Client({
      authStrategy: new LocalAuth({
        clientId: userId,
        dataPath: config.sessionDir,
      }),
      puppeteer: {
        headless: true,
        args: ["--no-sandbox", "--disable-setuid-sandbox", "--disable-gpu"],
      },
    });

    const session = { client, status: "connecting", qr: null };
    this.sessions.set(userId, session);

    // QR code event
    client.on("qr", (qr) => {
      session.qr = qr;
      session.status = "qr_pending";
      console.log(JSON.stringify({ level: "info", msg: "qr_generated", userId }));
    });

    // Authenticated (QR scanned, session saved)
    client.on("authenticated", () => {
      session.status = "authenticated";
      session.qr = null;
      console.log(JSON.stringify({ level: "info", msg: "authenticated", userId }));
    });

    // Ready (fully connected)
    client.on("ready", () => {
      session.status = "connected";
      console.log(JSON.stringify({ level: "info", msg: "session_ready", userId }));
      this.webhookSend("session_connected", { userId });
    });

    // Disconnected
    client.on("disconnected", (reason) => {
      session.status = "disconnected";
      console.log(JSON.stringify({ level: "warn", msg: "session_disconnected", userId, reason }));
      this.webhookSend("session_disconnected", { userId, reason });
      this.sessions.delete(userId);
    });

    // Incoming/outgoing messages
    client.on("message_create", (msg) => {
      this._handleMessage(userId, msg);
    });

    await client.initialize();
    return session;
  }

  /**
   * Handle a message event — push to backend via webhook.
   */
  async _handleMessage(userId, msg) {
    // Skip group messages, status broadcasts, and non-chat messages
    if (msg.isGroupMsg || msg.from === "status@broadcast") return;

    const isFromMe = msg.fromMe;
    const remotePhone = isFromMe ? msg.to : msg.from;

    this.webhookSend("message_received", {
      userId,
      messageId: msg.id._serialized,
      remotePhone,
      direction: isFromMe ? "outbound" : "inbound",
      body: msg.body || null,
      timestamp: msg.timestamp,
      type: msg.type,
    });
  }

  /**
   * Get the current QR code string for a session.
   */
  getQr(userId) {
    const session = this.sessions.get(userId);
    if (!session) return null;
    return session.qr;
  }

  /**
   * Get the status of a session.
   */
  getStatus(userId) {
    const session = this.sessions.get(userId);
    if (!session) return "disconnected";
    return session.status;
  }

  /**
   * Fetch contacts from a connected session.
   */
  async getContacts(userId) {
    const session = this.sessions.get(userId);
    if (!session || session.status !== "connected") {
      throw new Error("Session not connected");
    }

    const contacts = await session.client.getContacts();
    return contacts
      .filter((c) => !c.isGroup && !c.isMe && c.isWAContact)
      .map((c) => ({
        phone: c.id._serialized,
        name: c.pushname || c.name || null,
        shortName: c.shortName || null,
      }));
  }

  /**
   * Fetch recent chat messages for backfill.
   * Streams results to backend via webhook in batches.
   */
  async backfill(userId, { daysBack = 30, batchSize = 50 } = {}) {
    const session = this.sessions.get(userId);
    if (!session || session.status !== "connected") {
      throw new Error("Session not connected");
    }

    const chats = await session.client.getChats();
    const cutoff = Date.now() / 1000 - daysBack * 86400;
    let totalMessages = 0;

    for (const chat of chats) {
      if (chat.isGroup) continue;

      const messages = await chat.fetchMessages({ limit: 200 });
      const batch = [];

      for (const msg of messages) {
        if (msg.timestamp < cutoff) continue;

        batch.push({
          messageId: msg.id._serialized,
          remotePhone: chat.id._serialized,
          direction: msg.fromMe ? "outbound" : "inbound",
          body: msg.body || null,
          timestamp: msg.timestamp,
          type: msg.type,
        });

        if (batch.length >= batchSize) {
          await this.webhookSend("backfill_batch", { userId, messages: batch.splice(0) });
        }
      }

      if (batch.length > 0) {
        await this.webhookSend("backfill_batch", { userId, messages: batch });
      }

      totalMessages += messages.filter((m) => m.timestamp >= cutoff).length;
    }

    await this.webhookSend("backfill_complete", { userId, totalMessages });
    return { totalMessages };
  }

  /**
   * Destroy a session and clean up resources.
   */
  async destroySession(userId) {
    const session = this.sessions.get(userId);
    if (!session) return;

    try {
      await session.client.destroy();
    } catch (err) {
      console.error(JSON.stringify({ level: "error", msg: "destroy_failed", userId, error: err.message }));
    }
    this.sessions.delete(userId);
  }

  /**
   * Check if a session exists and is connected.
   */
  isConnected(userId) {
    const session = this.sessions.get(userId);
    return session?.status === "connected";
  }
}

module.exports = SessionManager;

def _verify_signature(body: bytes, signature: str) -> bool:
    """Verify HMAC-SHA256 signature from the sidecar."""
    expected = hmac.new(
        settings.WHATSAPP_WEBHOOK_SECRET.encode(),
        body,
        hashlib.sha256,
    ).hexdigest()
    return hmac.compare_digest(expected, signature)

async def whatsapp_webhook(
    request: Request,
    db: AsyncSession = Depends(get_db),

    body = await request.body()
    signature = request.headers.get("X-Webhook-Signature", "")

    if not _verify_signature(body, signature):
        raise HTTPException(status_code=401, detail="Invalid webhook signature")

    payload = json.loads(body)
    event = payload.get("event")
    data = payload.get("data", {})

    elif event == "session_disconnected":

    elif event == "message_received":

    elif event == "backfill_batch":

    elif event == "backfill_complete":
        await _handle_backfill_complete(data, db)
    else:
        logger.warning("whatsapp_webhook: unknown event %s", event)

    await db.commit()
    return {"received": True}

"""WhatsApp Celery tasks — backfill and session health checks."""
import uuid

from celery import shared_task
from sqlalchemy import select

from app.core.database import task_session
from app.models.user import User
from app.services.task_jobs.common import _run, logger, notify_sync_failure


@shared_task(
    name="app.services.tasks.sync_whatsapp_backfill",
    bind=True,
    max_retries=2,
    soft_time_limit=600,
    time_limit=900,
)
def sync_whatsapp_backfill(self, user_id: str) -> dict:
    """Trigger a WhatsApp message backfill via the sidecar."""

    async def _backfill(uid: uuid.UUID) -> dict:
        from app.integrations.whatsapp import trigger_backfill
        from app.services.sync_history import record_sync_start, record_sync_complete, record_sync_failure

        async with task_session() as db:
            result = await db.execute(select(User).where(User.id == uid))
            user = result.scalar_one_or_none()
            if not user:
                return {"status": "user_not_found"}

            if not user.whatsapp_connected:
                return {"status": "not_connected"}

            sync_event = await record_sync_start(uid, "whatsapp", "manual", db)

            try:
                backfill_result = await trigger_backfill(str(uid))
            except Exception as exc:
                await record_sync_failure(sync_event, str(exc), db=db)
                await db.commit()
                raise

            total = backfill_result.get("totalMessages", 0)
            await record_sync_complete(
                sync_event,
                records_created=total,
                details=backfill_result,
                db=db,
            )
            await db.commit()

        return {"status": "ok", "total_messages": total}

    try:
        uid = uuid.UUID(user_id)
    except ValueError:
        return {"status": "invalid_user_id"}

    try:
        return _run(_backfill(uid))
    except Exception as exc:
        logger.exception("sync_whatsapp_backfill failed for %s", user_id)
        if self.request.retries >= self.max_retries:
            notify_sync_failure.delay(user_id, "WhatsApp backfill", str(exc))
        raise self.retry(exc=exc, countdown=120) from exc


@shared_task(
    name="app.services.tasks.check_whatsapp_sessions",
    soft_time_limit=120,
    time_limit=180,
)
def check_whatsapp_sessions() -> dict:
    """Daily health check — verify active WhatsApp sessions are still connected."""

    async def _check() -> dict:
        from app.integrations.whatsapp import get_status
        from app.models.notification import Notification

        dead_sessions = 0

        async with task_session() as db:
            result = await db.execute(
                select(User).where(User.whatsapp_connected.is_(True))
            )
            users = result.scalars().all()

            for user in users:
                try:
                    status = await get_status(str(user.id))
                except Exception:
                    logger.warning(
                        "whatsapp health check: sidecar unreachable for user %s",
                        user.id,
                        exc_info=True,
                    )
                    status = "disconnected"

                if status != "connected":
                    user.whatsapp_connected = False
                    dead_sessions += 1
                    db.add(Notification(
                        user_id=user.id,
                        notification_type="sync",
                        title="WhatsApp disconnected",
                        body="Your WhatsApp session has expired. Please re-link in Settings.",
                        link="/settings",
                    ))

            await db.commit()

        return {"checked": len(users), "dead_sessions": dead_sessions}

    return _run(_check())
```

```typescript
"use client";

import { useState, useEffect, useRef, useCallback } from "react";
import { client } from "@/lib/api-client";
import type { SyncState } from "./use-settings-controller";

export type WhatsAppStep = "idle" | "qr_pending" | "connected" | "error";

export interface UseWhatsAppConnectFlowProps {
  whatsappConnect: SyncState;
  setWhatsappConnect: (s: SyncState) => void;
  onSuccess: () => Promise<void>;
}

export interface UseWhatsAppConnectFlowReturn {
  step: WhatsAppStep;
  qrData: string | null;
  error: string | null;
  startConnect: () => Promise<void>;
  cancel: () => void;
}

export function useWhatsAppConnectFlow({
  whatsappConnect,
  setWhatsappConnect,
  onSuccess,
}: UseWhatsAppConnectFlowProps): UseWhatsAppConnectFlowReturn {
  const [step, setStep] = useState<WhatsAppStep>("idle");
  const [qrData, setQrData] = useState<string | null>(null);
  const [error, setError] = useState<string | null>(null);
  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null);

  const stopPolling = useCallback(() => {
    if (pollRef.current) {
      clearInterval(pollRef.current);
      pollRef.current = null;
    }
  }, []);

  const startConnect = useCallback(async () => {
    setStep("qr_pending");
    setError(null);
    setWhatsappConnect({ status: "loading", message: "Starting WhatsApp session..." });

    const { data, error: err } = await client.POST("/api/v1/auth/whatsapp/connect", {});

    if (err) {
      setStep("error");
      setError("Failed to start WhatsApp session");
      setWhatsappConnect({ status: "error", message: "Connection failed" });
      return;
    }

    const result = data?.data as { status: string; qr: string | null } | undefined;
    if (result?.qr) {
      setQrData(result.qr);
    }

    setWhatsappConnect({ status: "idle", message: "Scan QR code with your phone" });

    // Start polling for status/QR updates
    pollRef.current = setInterval(async () => {
      const { data: statusData } = await client.GET("/api/v1/auth/whatsapp/qr", {});
      const s = statusData?.data as { status: string; qr: string | null } | undefined;

      if (s?.qr && s.qr !== qrData) {
        setQrData(s.qr);
      }

      if (s?.status === "connected") {
        stopPolling();
        setStep("connected");
        setWhatsappConnect({ status: "success", message: "WhatsApp connected!" });
        await onSuccess();
      }
    }, 3000);
  }, [setWhatsappConnect, onSuccess, stopPolling, qrData]);

  const cancel = useCallback(() => {
    stopPolling();
    setStep("idle");
    setQrData(null);
    setError(null);
    setWhatsappConnect({ status: "idle", message: "" });
  }, [stopPolling, setWhatsappConnect]);

  // Cleanup on unmount
  useEffect(() => {
    return () => stopPolling();
  }, [stopPolling]);

  return { step, qrData, error, startConnect, cancel };
}
```
